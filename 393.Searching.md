# B. Searching
- [Codeforces](https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/B)
## Code
```
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    long long x;
    cin >> n;
    vector <long long> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    cin >> x;
    int f = -1;
    for (int i = 0; i < n; i++) {
        if (a[i] == x) {
            f = i;
            break;
        }
    }
    cout << f;
}
```
