# AabBcCDd
- [Codechef](https://www.codechef.com/START224D/problems/AABBCCDD)
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int t; 
    cin >> t;
    while (t--) {
        int n; 
        string s;
        cin >> n >> s;
        vector<int> freq(26);
        for (auto ch : s) {
            char lc = tolower(ch);
            if (lc >= 'a' && lc <= 'z') freq[lc- 'a']++;
        }
        int m1 = 0, m2 = 0;
        for (auto i : freq) {
            if (i > m1) { 
                m2 = m1; 
                m1 = i; 
            }
            else if (i > m2) m2 = i; 
        }
        cout << (m1 + m2) << endl;
    }
}
```
