# E. Lowest number
- [Codeforces](https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/E)
## Code
```
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector <int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    int mini = INT_MAX, ind = -1;
    for (int i = 0; i < n; i++) {
        if (mini > a[i]) {
            mini = a[i];
            ind = i;
        }
    }
    cout << mini << " " << ind + 1;
}
```
