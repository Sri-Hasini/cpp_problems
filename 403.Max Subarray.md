# L. Max Subarray
- [Codeforces](https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/L)
## Code
```
#include <bits/stdc++.h>
using namespace std;
int main(){
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector <int> a(n);
        for (int i = 0; i < n; i++) cin >> a[i];

        for (int i = 0; i < n; i++) {
            int maxi = a[i];
            for (int j = i; j < n; j++) {
                maxi = max(maxi, a[j]);
                cout << maxi << " ";
            }
        }
        cout << endl;
    }
}
```
