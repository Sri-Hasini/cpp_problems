# 1260. Shift 2D Grid
- [Leetcode](https://leetcode.com/problems/shift-2d-grid/description/?envType=daily-question&envId=2026-07-20)
## Code 
```cpp
class Solution {
public:
    vector<vector<int>> shiftGrid(vector<vector<int>>& grid, int k) {
        int m = grid.size(), n = grid[0].size();       
        k = k % (m * n);
        vector <vector <int>> a(m, vector <int>(n));
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int ne = (i * n + j + k) % (m * n);
                int i1 = ne / n;
                int j1 = ne % n;
                a[i1][j1] = grid[i][j];
            }
        } 
        return a;
    }
};
```
