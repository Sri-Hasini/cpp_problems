# I. Smallest Pair
- [Codeforces](https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/I)
## Code
```
#include <bits/stdc++.h>
using namespace std;
int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector <int> a(n);
        for (int i = 0; i < n; i++) cin >> a[i];
        int mini = INT_MAX, x = 0, y = 0;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int val = a[i] + a[j] + j - i;
                if (mini > val) {
                    mini = val;
                    x = a[i];
                    y = a[j];
                }
            }
        }
        cout << mini << endl;
    }
}
```
