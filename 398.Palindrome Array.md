# G. Palindrome Array
- [Codeforces](https://codeforces.com/group/MWSDmqGsZm/contest/219774/problem/G)
## Code
```
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    cin >> n;
    vector <int> a(n);
    for (int i = 0; i < n; i++) cin >> a[i];
    bool f = true;
    int i = 0, j = n - 1;
    while (i < j) {
        if (a[i] != a[j]) {
            f = false;
            break;
        }
        i++;
        j--;
    }
    cout << ((f) ? "YES" : "NO");
}
```
