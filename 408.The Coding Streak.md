# The Coding Streak
- [Codechef](https://www.codechef.com/START224D/problems/STREAK)
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        vector<int> a(n);
        for (int i = 0; i < n; i++) cin >> a[i];
        int b = 0, c = 0;
        for (auto x : a) {
            if (x > 0) {
                c++;
                b = max(b, c);
            } 
            else c = 0;
        }
        cout << b << endl;
    }
}
```
