## Code in cpp:
```cpp
#include <bits/stdc++.h>
using namespace std;
const int mod=998244353;
long long mod_pow(long long base,int expo){
    long long result = 1;
    for (int i=0;i<expo;i++){
        result=(result*base)%mod;
    }
    return result;
}
int main() {
	int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        if(n%2==1){
            cout<<mod_pow(2,n-1)<<'\n';
        }
        else{
            cout<<(3*mod_pow(2,n-2))%mod<<'\n';
        }
    }
    return 0;
}
```
