# Accommodation
- [Codechef](https://www.codechef.com/START224D/problems/ACMDT)
```cpp
#include <bits/stdc++.h>
using namespace std;
using int_ = long long;

int main() {
    int t;
    cin >> t;
    while (t--) {
        int_ b, g, x, y, n;
        cin >> b >> g >> x >> y >> n;
        if (x + y > n) {
            cout << -1 << endl;
            continue;
        }
        int_ tot = b + g;
        int_ low = (tot + n - 1) / n;
        int_ hb = b / x;
        int_ hg = g / y;
        int_ high = min(hb, hg);
        if (low <= high) cout << low << endl;
        else cout << -1 << endl;
    }
}
```
